async function startAI() { const p = document.getElementById('prompt').value; const status = document.getElementById('upload-text'); status.innerText = "Thinking..."; try { const response = await fetch('/api/generate', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ prompt: p }) }); const result = await response.json(); // AI ka jawab screen par dikhane ke liye alert("AI Idea: " + result.candidates[0].content.parts[0].text); status.innerText = "Generation Complete!"; } catch (e) { alert("Error: API connect nahi ho saki."); status.innerText = "Error!"; } }
